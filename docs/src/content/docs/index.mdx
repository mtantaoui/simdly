## Quick Example

See Simdly in action with a simple vector addition:

<Tabs>
<TabItem label="Scalar (Slow)">

```rust
fn add_vectors(a: &[f32], b: &[f32]) -> Vec<f32> {
    a.iter().zip(b.iter()).map(|(x, y)| x + y).collect()
}
```

</TabItem>
<TabItem label="SIMD (Fast)">

```rust
use Simdly::f32x8;

fn add_vectors_simd(a: &[f32], b: &[f32]) -> Vec<f32> {
    let mut result = Vec::with_capacity(a.len());
    
    for (chunk_a, chunk_b) in a.chunks_exact(8).zip(b.chunks_exact(8)) {
        let va = f32x8::load_unaligned(chunk_a);
        let vb = f32x8::load_unaligned(chunk_b);
        let sum = va + vb;
        sum.store_unaligned(&mut result[..8]);
    }
    
    result
}
```

</TabItem>
</Tabs>
